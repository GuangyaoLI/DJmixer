149  static inline std::vector<std::string>
150  split(const std::string& str, const std::string& delims=" \t")
151  {
152  std::vector<std::string> res;
153  size_t init = 0;
154 
155  while (true) {
156  auto pos = str.find_first_of(delims, init);
157  if (pos == std::string::npos) {
158  res.emplace_back(str.substr(init, str.length()));
159  break;
160  }
161  res.emplace_back(str.substr(init, pos - init));
162  pos++;
163  init = pos;
164  }
165

273  static inline
274  int read_atmost_n(int fd, char* buf, size_t read_upto)
275  {
276  int rbytes = 0;
277  int eintr_cnter = 0;
278 
279  while (1) {
280  int read_bytes = read(fd, buf + rbytes, read_upto - rbytes);
281  if (read_bytes == -1) {
282  if (errno == EINTR) {
283  if (eintr_cnter >= 50) return -1;
284  eintr_cnter++;
285  continue;
286  }
287  return -1;
288  }
289  if (read_bytes == 0) return rbytes;
290 
291  rbytes += read_bytes;
292  }
293  return rbytes;
294  }

310  static inline int read_all(int fd, std::vector<char>& buf)
311  {
312  auto buffer = buf.data();
313  int total_bytes_read = 0;
314  int fill_sz = buf.size();
315 
316  while (1) {
317  const int rd_bytes = read_atmost_n(fd, buffer, fill_sz);
318 
319  if (rd_bytes == -1) { // Read finished
320  if (total_bytes_read == 0) return -1;
321  break;
322 
323  } else if (rd_bytes == fill_sz) { // Buffer full
324  const auto orig_sz = buf.size();
325  const auto new_sz = orig_sz * 2;
326  buf.resize(new_sz);
327  fill_sz = new_sz - orig_sz;
328 
329  //update the buffer pointer
330  buffer = buf.data();
331  total_bytes_read += rd_bytes;
332  buffer += total_bytes_read;
333 
334  } else { // Partial data ? Continue reading
335  total_bytes_read += rd_bytes;
336  fill_sz -= rd_bytes;
337  break;
338  }
339  }
340  buf.erase(buf.begin()+total_bytes_read, buf.end()); // remove extra nulls
341  return total_bytes_read;
342  }

1280  inline void Child::execute_child() {
1281  int sys_ret = -1;
1282  auto& stream = parent_->stream_;
1283 
1284  try {
1285  if (stream.write_to_parent_ == 0)
1286  stream.write_to_parent_ = dup(stream.write_to_parent_);
1287 
1288  if (stream.err_write_ == 0 || stream.err_write_ == 1)
1289  stream.err_write_ = dup(stream.err_write_);
1290 
1291  // Make the child owned descriptors as the
1292  // stdin, stdout and stderr for the child process
1293  auto _dup2_ = [](int fd, int to_fd) {
1294  if (fd == to_fd) {
1295  // dup2 syscall does not reset the
1296  // CLOEXEC flag if the descriptors
1297  // provided to it are same.
1298  // But, we need to reset the CLOEXEC
1299  // flag as the provided descriptors
1300  // are now going to be the standard
1301  // input, output and error
1302  util::set_clo_on_exec(fd, false);
1303  } else if(fd != -1) {
1304  int res = dup2(fd, to_fd);
1305  if (res == -1) throw OSError("dup2 failed", errno);
1306  }
1307  };
1308 
1309  // Create the standard streams
1310  _dup2_(stream.read_from_parent_, 0); // Input stream
1311  _dup2_(stream.write_to_parent_, 1); // Output stream
1312  _dup2_(stream.err_write_, 2); // Error stream
1313 
1314  // Close the duped descriptors
1315  if (stream.read_from_parent_ != -1 && stream.read_from_parent_ > 2)
1316  close(stream.read_from_parent_);
1317 
1318  if (stream.write_to_parent_ != -1 && stream.write_to_parent_ > 2)
1319  close(stream.write_to_parent_);
1320 
1321  if (stream.err_write_ != -1 && stream.err_write_ > 2)
1322  close(stream.err_write_);
1323 
1324  // Close all the inherited fd's except the error write pipe
1325  if (parent_->close_fds_) {
1326  int max_fd = sysconf(_SC_OPEN_MAX);
1327  if (max_fd == -1) throw OSError("sysconf failed", errno);
1328 
1329  for (int i = 3; i < max_fd; i++) {
1330  if (i == err_wr_pipe_) continue;
1331  close(i);
1332  }
1333  }
1334 
1335  // Change the working directory if provided
1336  if (parent_->cwd_.length()) {
1337  sys_ret = chdir(parent_->cwd_.c_str());
1338  if (sys_ret == -1) throw OSError("chdir failed", errno);
1339  }
1340 
1341  if (parent_->has_preexec_fn_) {
1342  parent_->preexec_fn_();
1343  }
1344 
1345  if (parent_->session_leader_) {
1346  sys_ret = setsid();
1347  if (sys_ret == -1) throw OSError("setsid failed", errno);
1348  }
1349 
1350  // Replace the current image with the executable
1351  if (parent_->env_.size()) {
1352  for (auto& kv : parent_->env_) {
1353  setenv(kv.first.c_str(), kv.second.c_str(), 1);
1354  }
1355  sys_ret = execvp(parent_->exe_name_.c_str(), parent_->cargv_.data());
1356  } else {
1357  sys_ret = execvp(parent_->exe_name_.c_str(), parent_->cargv_.data());
1358  }
1359 
1360  if (sys_ret == -1) throw OSError("execve failed", errno);
1361 
1362  } catch (const OSError& exp) {
1363  // Just write the exception message
1364  // TODO: Give back stack trace ?
1365  std::string err_msg(exp.what());
1366  //ATTN: Can we do something on error here ?
1367  util::write_n(err_wr_pipe_, err_msg.c_str(), err_msg.length());

1369  }
1370

1372  // exit failure
1373  exit (EXIT_FAILURE);
